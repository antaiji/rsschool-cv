function mergeArrays(arr1, arr2) {
  const mySet = new Set(
    [...arr1, ...arr2].sort((a, b) => {
      if (a > b) return 1;
      if (a < b) return -1;
      return 0;
    })
  );

  return Array.from(mySet);
}
